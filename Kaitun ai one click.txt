-- ===== TỰ ĐỘNG CHỌN TEAM (HẢI QUÂN) =====
task.spawn(function()
    local p = game.Players.LocalPlayer
    repeat task.wait() until p and p:FindFirstChild("PlayerGui")
    local pGui = p.PlayerGui
    local chooseTeam = pGui:FindFirstChild("Main (minimal)") and pGui["Main (minimal)"]:FindFirstChild("ChooseTeam")
    if chooseTeam then
        local attempts = 0
        repeat
            task.wait(0.5)
            attempts = attempts + 1
            if chooseTeam.Visible then
                pcall(function()
                    game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("CommF_"):InvokeServer("SetTeam", "Marines")
                    print("[Auto Team] Đã chọn Hải quân")
                end)
                break
            end
        until p.Team ~= nil or attempts >= 20
    end
end)

-- ===== ANTI-LAG & LOW GRAPHICS =====
task.spawn(function()
    repeat task.wait() until game:IsLoaded() and game.Players.LocalPlayer and game.Players.LocalPlayer.Character
    pcall(function()
        local Lighting = game:GetService("Lighting")
        Lighting.GlobalShadows = false
        Lighting.FogEnd = 100000
        Lighting.Brightness = 1
        Lighting.ClockTime = 12
        local Terrain = workspace.Terrain
        Terrain.WaterWaveSize, Terrain.WaterWaveSpeed, Terrain.WaterReflectance, Terrain.WaterTransparency = 0, 0, 0, 0
        for _, obj in pairs(workspace:GetDescendants()) do
            if obj:IsA("ParticleEmitter") or obj:IsA("Fire") or obj:IsA("Smoke") or obj:IsA("Trail") then
                obj.Enabled = false
            end
            if obj:IsA("BasePart") then
                obj.Material = Enum.Material.Plastic
                obj.Reflectance = 0
                obj.CastShadow = false
            end
            if obj:IsA("Decal") or obj:IsA("Texture") then
                obj.Texture = ""
                obj.Transparency = 1
            end
        end
        for _, effect in pairs(Lighting:GetDescendants()) do
            if effect:IsA("PostEffect") or (effect:IsA("BlurEffect") and effect.Name ~= "CustomBlur") then
                effect.Enabled = false
            end
        end
        local player = game.Players.LocalPlayer
        if player.PlayerScripts:FindFirstChild("NewIslandLOD") then player.PlayerScripts.NewIslandLOD:Destroy() end
        if player.PlayerScripts:FindFirstChild("IslandLOD") then player.PlayerScripts.IslandLOD:Destroy() end
        local ReplicatedStorage = game:GetService("ReplicatedStorage")
        local container = ReplicatedStorage:FindFirstChild("Container", true)
        if container then
            for _, name in ipairs({"AirDash", "LightningTP", "Damage", "Confetti", "LevelUp"}) do
                local target = container:FindFirstChild(name, true)
                if target then target:Destroy() end
            end
        end
        if setfps then setfps(120) end
    end)
end)

-- ===== ANTI-3TN =====
local function kill3TN(o)
    if o and o.Name == "3TN" then o:Destroy() end
end

local function createFloatingToggle()
    if floatingToggle and floatingToggle.Parent then return end
    floatingToggle = Instance.new("ImageButton")
    floatingToggle.BackgroundColor3 = Color3.fromRGB(170,130,255)
    floatingToggle.BackgroundTransparency = 0.2

    floatingToggle.AutoButtonColor = false
    floatingToggle.ZIndex = 999
    floatingToggle.Parent = screenGui
    local floatCorner = Instance.new("UICorner")
    floatCorner.CornerRadius = UDim.new(1,0)
    floatCorner.Parent = floatingToggle
    local floatStroke = Instance.new("UIStroke")
    floatStroke.Color = Color3.fromRGB(80,80,100)
    floatStroke.Thickness = 1.5
    floatStroke.Parent = floatingToggle

    local draggingFloat = false
    local dragStartFloat, startPosFloat
    floatingToggle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            draggingFloat = true
            dragStartFloat = input.Position
            startPosFloat = floatingToggle.Position
        end
    end)
    floatingToggle.InputEnded:Connect(function() draggingFloat = false end)
    UserInputService.InputChanged:Connect(function(input)
        if draggingFloat and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            local delta = input.Position - dragStartFloat
            floatingToggle.Position = UDim2.new(startPosFloat.X.Scale, startPosFloat.X.Offset + delta.X,
                startPosFloat.Y.Scale, startPosFloat.Y.Offset + delta.Y)
        end
    end)

    floatingToggle.MouseButton1Click:Connect(function()
        uiVisible = true
        mainFrame.Visible = true
        blur.Size = 6
        toggleBtn.Image = "rbxassetid://105304056374139"
        floatingToggle:Destroy()
        floatingToggle = nil
    end)
end

    -- Hàm cập nhật thông tin
    local function updateStats()
        local p = game.Players.LocalPlayer
        if not p then return end
        local data = p:FindFirstChild("Data")
        local leaderstats = p:FindFirstChild("leaderstats")
        local level = 0
        local beli = 0
        local bounty = 0
        local frag = 0
        local race = "Unknown"

        if data then
            local lv = data:FindFirstChild("Level")
            if lv then level = lv.Value end
            local b = data:FindFirstChild("Beli")
            if b then beli = b.Value end
            local f = data:FindFirstChild("Fragments")
            if f then frag = f.Value end
            local r = data:FindFirstChild("Race")
            if r then race = r.Value end
        end
        if leaderstats then
            local bo = leaderstats:FindFirstChild("Bounty/Honor")
            if bo then bounty = bo.Value end
        end

        levelVal.Text = tostring(level)
        beliVal.Text = tostring(beli)
        bountyVal.Text = tostring(bounty)
        fragVal.Text = tostring(frag)
        levelLabel.Text = "Level: " .. tostring(level)
        raceLabel.Text = "Race: " .. race

        -- Cập nhật session time
        local sessionStart = getgenv().KaitunSessionStart or os.time()
        local elapsed = os.time() - sessionStart
        local h = math.floor(elapsed / 3600)
        local m = math.floor((elapsed % 3600) / 60)
        local s = elapsed % 60
        sessionLabel.Text = string.format("Session: %02d:%02d:%02d", h, m, s)
    end
